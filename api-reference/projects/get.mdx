# Get Project Details

Get detailed information about a specific project by its UUID.

## API Request

<Card title="GET Project Detail" icon="building">
  ```bash
  GET /api/v1/initiatives/projects/{uuid}
  ```
</Card>

## Path Parameters

| Parameter | Type | Description | Required |
|-----------|------|-------------|----------|
| uuid | String | The UUID of the project | Yes |

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  Project details
  <Expandable title="Properties">
    <ResponseField name="uuid" type="string">
      Unique identifier for the project
    </ResponseField>
    <ResponseField name="name" type="string">
      Project name in English
    </ResponseField>
    <ResponseField name="name_dv" type="string">
      Project name in Dhivehi
    </ResponseField>
    <ResponseField name="function" type="string">
      Project function
    </ResponseField>
    <ResponseField name="agency" type="object">
      Agency information
      <Expandable title="Properties">
        <ResponseField name="name" type="string">Agency name in English</ResponseField>
        <ResponseField name="name_dv" type="string">Agency name in Dhivehi</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="project" type="object">
      Project details
      <Expandable title="Properties">
        <ResponseField name="description" type="string">Project description</ResponseField>
        <ResponseField name="start_date" type="string">Project start date</ResponseField>
        <ResponseField name="end_date" type="string">Project end date</ResponseField>
        <ResponseField name="locations" type="array">Project locations</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="status" type="object">
      Project status
      <Expandable title="Properties">
        <ResponseField name="name" type="string">Status name</ResponseField>
        <ResponseField name="slug" type="string">Status identifier</ResponseField>
        <ResponseField name="color" type="string">Status color code</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="contracts" type="array">
      Project contracts
      <Expandable title="Properties">
        <ResponseField name="uuid" type="string">Contract UUID</ResponseField>
        <ResponseField name="contract_number" type="string">Contract number</ResponseField>
        <ResponseField name="contract_name" type="string">Contract name</ResponseField>
        <ResponseField name="vendor" type="object">Vendor information</ResponseField>
        <ResponseField name="current_scope" type="string">Current contract scope</ResponseField>
        <ResponseField name="commencement_date" type="string">Contract start date</ResponseField>
        <ResponseField name="scheduled_completion_date" type="string">Scheduled completion date</ResponseField>
        <ResponseField name="actual_completion_date" type="string">Actual completion date</ResponseField>
        <ResponseField name="status" type="string">Contract status</ResponseField>
        <ResponseField name="current_contract_value" type="number">Current contract value</ResponseField>

    <ResponseField name="phases" type="array">
      Project phases

  </Expandable>
